.MODE       Register    ; Sets the VM use the register instruction set.
.HEAP       4096        ; Start with 4 KiB heap size
.HEAP_MAX   67 108 864  ; 64 MiB. Overflows will terminate the program at runtime.

// Execution starts here.
      JMP     init                  // Start by jumping to init.
      JMP     nonsense;asd 

init:
      NOP	                          ; Do nothing.
      NOP
      NOP	                          // Do nothing.
      JMP	    main                  ; Jump to main.

main:  // Entry point.
  ; It starts here.
      LDI     R0,     8 + 2         // Addition
      LDI     R0,     8 - 2         // Subtraction
      LDI     R0,     8 * 2 + 8 / 4// Multiplication
      LDI     R0,     8 / 2 / 2     // Division.
      LDI     R0,     8 / 4         ; One more division.
      LDI     R0,     1024          ; Load 1024 into R0.
      LDI     R1,     6             // Load 6 into R1.
      ADD     R0,     R1,     R2
      SUB     R0,     R1,     R2    ; Subtract
